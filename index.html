1. 基本語法介紹

  • 縮排原則：用 4 個空格縮排表示程式區塊，不要混合 tab 與空格。

  • 常用內建函式：`print()`、`input()`、`len()`、`type()`、`range()`、`sum()` 等。

5.4 函式修飾器 (Decorators)

# 修飾器本質：接收函式、回傳包裝後的新函式
# 實務上建議搭配 functools.wraps 保留原函式名稱與 docstring
import functools

# 基本修飾器
def debug_decorator(func):
    """印出函式呼叫和回傳值的修飾器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"呼叫函式 {func.__name__}，參數: {args}, {kwargs}")
        result = func(*args, **kwargs)
        print(f"函式 {func.__name__} 回傳: {result}")
        return result
    return wrapper

@debug_decorator
def add(x, y):
    return x + y

@debug_decorator
def multiply(x, y):
    return x * y

add(3, 4)
multiply(5, 6)

# 計時修飾器
import time
def timing_decorator(func):
    """測量函式執行時間的修飾器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} 執行時間: {end - start:.4f} 秒")
        return result
    return wrapper

@timing_decorator
def slow_function():
    time.sleep(0.5)
    return "Done"

slow_function()

OOP 核心觀念速查

概念 用途 你要記得的事
`__init__` 建立物件時初始化資料 第一個參數永遠是 `self`
Instance Method 操作單一物件自己的資料 最常用，第一個參數是 `self`
Class Method 操作整個 class 共用狀態 用 `@classmethod`，第一個參數是 `cls`
Static Method 邏輯相關但不依賴物件狀態 用 `@staticmethod`，沒有 `self` / `cls`
Inheritance 讓子類別重用父類別邏輯 用 `super()` 取用父類別初始化或方法
Encapsulation 限制外部直接亂改內部狀態 Python 常用 `_name` 與 method 包裝操作

    def __init__(self, name, email):
        self.name = name
        self.email = email
        self._login_count = 0
        User.total_users += 1

        self._login_count += 1
        return f"{self.name} logged in"

    def profile(self):
        return {
            "name": self.name,
            "email": self.email,
            "login_count": self._login_count,
        }

    @classmethod
    def how_many_users(cls):
        return cls.total_users

    @staticmethod
    def is_valid_email(email):

        super().__init__(name, email)
        self.role = role

    def profile(self):
        data = super().profile()
        data["role"] = self.role
        return data

這段 OOP 範例要怎樣讀

  2. `name`、`email`、`_login_count` 是 attribute。

  4. `how_many_users()` 是 class method，操作 class 共用資料。

  6. `AdminUser(User)` 表示繼承，子類別能重用父類別功能。

    with open(config_file, "r", encoding="utf-8") as f:
        config = yaml.safe_load(f)

    if platform not in config.get("platforms", {}):
        raise ValueError(f"不支援的平台: {platform}")

    capabilities = config["platforms"][platform].copy()
    capabilities.update(config.get("common", {}))
    return capabilities

def scroll_to_element(driver, locator, direction="down"):
    """
    滑動頁面直到指定元素可見
    Args:
        driver: Appium Driver 實例
        locator: 元素定位器 (By, value)
        direction: 滑動方向 ("down" 或 "up")
    Returns:
        WebElement: 找到的元素
    """
    # 獲取裝置螢幕尺寸
    screen_size = driver.get_window_size()
    screen_width = screen_size['width']
    screen_height = screen_size['height']
    
    # 計算滑動起點和終點座標
    start_x = int(screen_width / 2)
    end_x = start_x
    
    if direction == "down":
        start_y = int(screen_height * 0.8)  # 從底部 80% 位置開始
        end_y = int(screen_height * 0.2)    # 滑動到頂部 20% 位置
    else:
        start_y = int(screen_height * 0.2)  # 從頂部 20% 位置開始
        end_y = int(screen_height * 0.8)    # 滑動到底部 80% 位置
    
    # 滑動並查找元素，最多嘗試 5 次
    for _ in range(5):
        try:
            element = driver.find_element(*locator)
            # 如果元素可見，返回該元素
            if element.is_displayed():
                return element
        except:
            pass
        
        # 執行滑動操作
        driver.swipe(start_x, start_y, end_x, end_y, duration=800)
        # 等待滑動動畫完成（不要用 implicitly_wait 當 sleep）
        import time
        time.sleep(0.5)
    
    raise Exception(f"無法找到元素: {locator}")

from appium.webdriver.common.appiumby import AppiumBy

class DatePicker:
    """處理日期選擇器操作"""
    
    def __init__(self, driver):
        self.driver = driver
        self.platform = driver.capabilities.get("platformName").lower()
    
    def select_date(self, year, month, day):
        """選擇日期
        Args:
            year: 年份
            month: 月份 (1-12)
            day: 日期 (1-31)
        """
        if self.platform == "android":
            self._select_date_android(year, month, day)
        elif self.platform == "ios":
            self._select_date_ios(year, month, day)
        else:
            raise Exception(f"不支援的平台: {self.platform}")
    
    def _select_date_android(self, year, month, day):
        """Android 日期選擇器操作"""
        # 定位到年份選擇器並選擇年份
        year_picker = WebDriverWait(self.driver, 10).until(
            EC.presence_of_element_located((AppiumBy.ID, "com.example.app:id/year_picker"))
        )
        year_picker.click()
        
        # 選擇年份（這裡假設是滾輪選擇器）
        target_year = str(year)
        while True:
            current_year = year_picker.text
            if current_year == target_year:
                break
            elif int(current_year) < year:
                # 向上滑動
                self._swipe_wheel_up(year_picker)
            else:
                # 向下滑動
                self._swipe_wheel_down(year_picker)
        
        # 定位到月份選擇器並選擇月份
        month_picker = WebDriverWait(self.driver, 10).until(
            EC.presence_of_element_located((AppiumBy.ID, "com.example.app:id/month_picker"))
        )
        month_picker.click()
        
        # 選擇月份
        target_month = str(month)
        while True:
            current_month = month_picker.text
            if current_month == target_month:
                break
            elif int(current_month) < month:
                self._swipe_wheel_up(month_picker)
            else:
                self._swipe_wheel_down(month_picker)
        
        # 定位到日期選擇器並選擇日期
        day_picker = WebDriverWait(self.driver, 10).until(
            EC.presence_of_element_located((AppiumBy.ID, "com.example.app:id/day_picker"))
        )
        day_picker.click()
        
        # 選擇日期
        target_day = str(day)
        while True:
            current_day = day_picker.text
            if current_day == target_day:
                break
            elif int(current_day) < day:
                self._swipe_wheel_up(day_picker)
            else:
                self._swipe_wheel_down(day_picker)
    
    def _select_date_ios(self, year, month, day):
        """iOS 日期選擇器操作"""
        # iOS 日期選擇器通常是 PickerWheel 類型
        try:
            # 定位到日期選擇器
            date_picker = WebDriverWait(self.driver, 10).until(
                EC.presence_of_element_located((AppiumBy.CLASS_NAME, "XCUIElementTypePickerWheel"))
            )
            
            # 設置日期格式（這取決於實際的日期選擇器格式）
            date_str = f"{month}/{day}/{year}"
            date_picker.send_keys(date_str)
            
            # 等待日期選擇器更新
            import time
            time.sleep(1)
        except Exception as e:
            print(f"iOS 日期選擇器操作失敗: {e}")
            # 如果標準方法失敗，嘗試手動滑動
            self._manual_date_select_ios(year, month, day)
    
    def _manual_date_select_ios(self, year, month, day):
        """手動滑動選擇日期（作為後備方法）"""
        # 定位到年、月、日選擇器
        pickers = self.driver.find_elements(AppiumBy.CLASS_NAME, "XCUIElementTypePickerWheel")
        
        if len(pickers) == 3:  # 假設有三個選擇器：年、月、日
            # 選擇年份
            self._select_value_ios_picker(pickers[0], str(year))
            # 選擇月份
            self._select_value_ios_picker(pickers[1], str(month))
            # 選擇日期
            self._select_value_ios_picker(pickers[2], str(day))
    
    def _select_value_ios_picker(self, picker, target_value):
        """選擇 iOS PickerWheel 的值"""
        current_value = picker.text
        while current_value != target_value:
            if int(current_value) < int(target_value):
                picker.send_keys("down")
            else:
                picker.send_keys("up")
            
            # 等待選擇器更新
            import time
            time.sleep(0.5)
            current_value = picker.text
    
    def _swipe_wheel_up(self, element):
        """向上滑動選擇器（Android）"""
        rect = element.rect
        center_x = rect['x'] + rect['width'] / 2
        start_y = rect['y'] + rect['height'] * 0.8
        end_y = rect['y'] + rect['height'] * 0.2
        self.driver.swipe(center_x, start_y, center_x, end_y, duration=500)
    
    def _swipe_wheel_down(self, element):
        """向下滑動選擇器（Android）"""
        rect = element.rect
        center_x = rect['x'] + rect['width'] / 2
        start_y = rect['y'] + rect['height'] * 0.2
        end_y = rect['y'] + rect['height'] * 0.8
        self.driver.swipe(center_x, start_y, center_x, end_y, duration=500)

class LoginPage:
    # 定位器字典：根據平台返回對應的定位器
    LOCATORS = {
        "login_button": {
            "android": (AppiumBy.ID, "com.demo:id/btn_login"),      # Android 平台使用 ID 定位
            "ios": (AppiumBy.ACCESSIBILITY_ID, "login_button"),    # iOS 平台使用 Accessibility ID 定位
        },
        "username_input": {
            "android": (AppiumBy.ID, "com.demo:id/input_username"), # Android 平台使用 ID 定位
            "ios": (AppiumBy.IOS_PREDICATE, 'name == "username_input"'),  # iOS 平台使用 Predicate 定位
        },
    }

    def __init__(self, driver):
        """初始化 LoginPage 類"""
        self.driver = driver                                      # 保存 Appium Driver 實例
        # 獲取當前平台名稱，並轉為小寫
        self.platform = driver.capabilities.get("platformName", "").lower()

    def locator(self, name):
        """根據元素名稱和當前平台返回定位器"""
        return self.LOCATORS[name][self.platform]

    def find(self, name):
        """根據元素名稱找到元素"""
        by, value = self.locator(name)                            # 獲取定位方式和值
        return self.driver.find_element(by, value)                 # 找到並返回元素

  • 測試腳本只寫 `find("login_button")`，不用到處寫平台判斷。
  • 定位差異集中管理，頁面改版時比較好維護。
  • Page Object 很適合和 Python OOP 一起學。

wait 的實戰建議

  • 少用固定 sleep()，優先用顯式等待。
  • presence：元素在 DOM 中；visibility：可見；clickable：可點擊。
  • 分清楚要等的是存在、可見，還是可點擊。
  • 如果畫面有 loading、toast、動畫，也要等它消失。

def tap_by_coordinates(driver, x, y):
    """使用座標點擊元素"""
    finger = PointerInput("touch", "finger")
    actions = ActionBuilder(driver, mouse=finger)
    actions.pointer_action.move_to_location(x, y)
    actions.pointer_action.pointer_down()

    actions.pointer_action.pointer_up()
    actions.perform()

def safe_tap_element(element, retries=3):
    """安全點擊元素，失敗時使用座標點擊作為備援"""
    for attempt in range(retries):
        try:
            element.click()
            return True
        except Exception as e:
            print(f"第 {attempt + 1} 次點擊失敗: {e}")
            if attempt < retries - 1:
                print("嘗試使用座標點擊")
                rect = element.rect
                center_x = int(rect["x"] + rect["width"] / 2)
                center_y = int(rect["y"] + rect["height"] / 2)
                tap_by_coordinates(driver, center_x, center_y)
                return True
    return False


# 使用安全點擊
if safe_tap_element(element):
    print("點擊成功")
else:
    print("所有點擊方法都失敗")

什麼時候需要使用座標點擊

  • 元素定位成功但 click() 失敗：這是最常見的場景，可能是因為元素被覆蓋、透明、或者 App 對 click 事件進行了特殊處理
  • 動畫或遮罩問題：當元素正在動畫中或被半透明遮罩覆蓋時，普通的 click() 可能會失敗
  • 客製化控件：某些應用有自己實現的控件（如自定義按鈕或選單），Appium 可能無法正確模擬點擊
  • 畫布或地圖類元素：這些元素可能沒有可識別的子元素，需要精確的座標點擊
  • 特殊交互場景：如雙擊、長按、多指操作等，普通的 click() 方法無法滿足需求

實戰建議

  • 永遠先嘗試使用標準的 element.click() 方法，只有在失敗時才使用座標點擊作為備援
  • 結合等待策略使用，確保元素真的可以點擊時再進行操作
  • 記錄點擊失敗的場景和解決方案，建立自己的問題排查手冊
  • 在 CI/CD 流程中添加點擊失敗的監控和報告機制

pytest 教學：從基礎到進階的測試架構

pytest 是 Python 中最受歡迎的測試框架之一，支援單元測試、整合測試、功能測試。 本章節詳細介紹 pytest 的基本用法、fixture、標記、參數化、以及如何結合 Appium 使用。

1. 為什麼選擇 pytest

  • 語法簡潔，使用 `assert` 即可，不需要特定的斷言方法
  • 強大的 fixture 系統，用於共享資源與設定
  • 豐富的外掛生態系統 (pytest-xdist、pytest-html 等)
  • 支援標記 (marker) 用於分類與選取測試
  • 整合性強，適合從單元測試到端對端測試

2. 文件與命名約定

  • 測試檔案：以 `test_` 開頭或 `_test` 結尾
  • 測試函式：以 `test_` 開頭
  • 測試類別：以 `Test` 開頭，且不能有 `__init__` 方法
  • 常見命名：`test_login.py`、`test_cart_functionality.py`

3. 安裝與執行

  • 安裝：`pip install pytest`
  • 執行所有測試：`pytest`
  • 執行特定檔案：`pytest test_login.py`
  • 執行特定函式：`pytest test_login.py::test_login_success`
  • 顯示詳細資訊：`pytest -v` 或 `pytest -v -s`

import yaml
import os

class ConfigManager:
    """配置管理器 - 用於加載和管理 YAML 配置文件"""
    
    def __init__(self, config_file="config/capabilities.yaml"):
        """初始化配置管理器
        
        Args:
            config_file: 配置文件路徑，預設為 config/capabilities.yaml
        """
        self.config_file = config_file
        self._config = None
        self._load_config()
    
    def _load_config(self):
        """載入配置文件"""
        try:
            with open(self.config_file, 'r', encoding='utf-8') as file:
                self._config = yaml.safe_load(file)
            print(f"成功載入配置文件: {self.config_file}")
        except FileNotFoundError:
            raise Exception(f"配置文件未找到: {self.config_file}")
        except yaml.YAMLError as e:
            raise Exception(f"配置文件格式錯誤: {e}")
        except Exception as e:
            raise Exception(f"載入配置文件時發生錯誤: {e}")
    
    def reload_config(self):
        """重新載入配置文件"""
        self._load_config()
    
    def get_platform_capabilities(self, platform):
        """獲取指定平台的 capabilities 配置
        
        Args:
            platform: 平台名稱 ("android" 或 "ios")
            
        Returns:
            dict: 平台的 capabilities 配置
        """
        if platform not in self._config.get('platforms', {}):
            raise Exception(f"不支援的平台: {platform}")
        
        # 合併平台特定配置和共用配置
        capabilities = self._config['platforms'][platform].copy()
        capabilities.update(self._config.get('common', {}))
        return capabilities
    
    def get_test_data(self, key_path):
        """獲取測試數據配置
        
        Args:
            key_path: 配置項的路徑，例如 "login.valid"
            
        Returns:
            配置項的值
        """
        keys = key_path.split('.')
        data = self._config.get('test_data', {})
        
        try:
            for key in keys:
                data = data[key]
            return data
        except KeyError:
            raise Exception(f"配置項未找到: {key_path}")
    
    def get_all_config(self):
        """獲取完整的配置字典"""
        return self._config
    
    def get_config_value(self, key_path, default=None):
        """獲取配置值
        
        Args:
            key_path: 配置項的路徑，例如 "common.language"
            default: 預設值，若配置項不存在則返回此值
            
        Returns:
            配置項的值或預設值
        """

        data = self._config
        
        try:
            for key in keys:
                data = data[key]
            return data
        except (KeyError, TypeError):
            return default

# 定位器管理
class LoginPageLocators:
    """登錄頁面定位器 - 根據平台返回對應的定位器"""
    LOCATORS = {
        "username_input": {
            "android": (AppiumBy.ID, "com.example.app:id/et_username"),  # Android 使用 ID 定位
            "ios": (AppiumBy.IOS_PREDICATE, 'name == "username_input" OR label == "username_input"')  # iOS 使用 Predicate 定位
        },
        "password_input": {
            "android": (AppiumBy.ID, "com.example.app:id/et_password"),  # Android 使用 ID 定位
            "ios": (AppiumBy.IOS_PREDICATE, 'name == "password_input" OR label == "password_input"')  # iOS 使用 Predicate 定位
        },
        "login_button": {
            "android": (AppiumBy.ID, "com.example.app:id/btn_login"),    # Android 使用 ID 定位
            "ios": (AppiumBy.ACCESSIBILITY_ID, "login_button")          # iOS 使用 Accessibility ID 定位
        },
        "welcome_text": {
            "android": (AppiumBy.ID, "com.example.app:id/tv_welcome"),  # Android 使用 ID 定位
            "ios": (AppiumBy.IOS_PREDICATE, 'name CONTAINS "Welcome"')  # iOS 使用 Predicate 定位，包含文本
        },
        "error_message": {
            "android": (AppiumBy.ID, "com.example.app:id/tv_error"),    # Android 使用 ID 定位
            "ios": (AppiumBy.IOS_PREDICATE, 'name CONTAINS "Error"')    # iOS 使用 Predicate 定位，包含文本
        }
    }
    
    @classmethod
    def get_locator(cls, element_name, platform):
        """根據元素名稱和平台獲取定位器"""
        if element_name not in cls.LOCATORS:
            raise ValueError(f"Element '{element_name}' not found in locators")  # 元素不存在時拋異常
        
        if platform not in cls.LOCATORS[element_name]:
            raise ValueError(f"Platform '{platform}' not supported for element '{element_name}'")  # 平台不支持時拋異常
        
        return cls.LOCATORS[element_name][platform]  # 返回定位器

Python Package Control：uv 與 poetry 詳細教學

Python 套件管理工具用於管理虛擬環境、依賴套件與專案結構。本章節詳細介紹 uv (快速輕巧) 與 poetry (完整管理) 兩種主流工具的安裝、設定與使用。

1. 為什麼需要套件管理工具

  • 管理虛擬環境，避免套件衝突
  • 追蹤依賴套件版本，確保開發與生產環境一致性
  • 簡化安裝、升級、移除套件的流程
  • 支援專案打包與發佈
  • 提高開發效率與程式碼品質

2. 傳統方法的問題

  • 直接使用 global pip 安裝會導致依賴衝突
  • virtualenv 操作複雜，需要手動啟動與停止
  • requirements.txt 格式不標準，依賴解析困難
  • 缺少 lock 文件，無法確保精確的依賴版本

3. 兩種主流工具比較

  • uv：快速、輕巧，適合小型專案與腳本
  • poetry：完整管理，適合大型專案與團隊開發
  • 兩者都支援 `pyproject.toml` 規範
  • 都提供虛擬環境管理與 lock 文件功能
